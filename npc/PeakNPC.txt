-	script	Highest Peak	-1,{

OnPCLoginEvent:

// Asignar el número actual de jugadores en línea
sleep 1000;
.onlineusers = getusers(1);

// Fecha de hoy
.date$ = gettime(DT_YEAR)+"-"+gettime(DT_MONTH)+"-"+gettime(DT_DAYOFMONTH);

// Consulta para el pico más alto en la base de datos
query_sql("SELECT `users` FROM `" + .sqltable$ + "` ORDER BY `users` DESC LIMIT 1",.@countusers);

if(getarraysize(.@countusers) == 0) {
	// No parece haber una fila en nuestra tabla, así que vamos a crear una
	query_sql("INSERT INTO `" + .sqltable$ + "` (`users`, `date`) VALUES (" + .onlineusers + ", '" + .date$ + "')");
} else {
	if(.onlineusers > .@countusers[0]) {
		// Aquí es donde verificamos si queremos anunciar un nuevo pico
		if(.displaypeakannounce == 1){
			announce "¡Hemos alcanzado un nuevo nivel de jugadores! Ahora tenemos "+.onlineusers+" en línea!",bc_all;
		}
		// Ahora vamos a actualizar la tabla con nuestro nuevo pico de jugadores
		query_sql("UPDATE `" + .sqltable$ + "` SET `users` = '" + .onlineusers + "', `date` = '" + .date$ + "'");
	}
}
end;

OnInit:
// CONFIGS
	// Primero necesitamos un nombre para la tabla SQL
	.sqltable$ = "ra_onlinepeak";
	
	// ¿Te gustaría un anuncio cuando alcances un pico más alto?
	// 1 = Sí
	// 0 = No
	.displaypeakannounce = 1;

// FIN DE CONFIGURACIONES

}
